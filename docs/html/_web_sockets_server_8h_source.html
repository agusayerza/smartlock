181  /*
182  * Called at client socket connect handshake negotiation time for each http header that is not
183  * a websocket specific http header (not Connection, Upgrade, Sec-WebSocket-*)
184  * If the custom httpHeaderValidationFunc returns false for any headerName / headerValue passed, the
185  * socket negotiation is considered invalid and the upgrade to websockets request is denied / rejected
186  * This mechanism can be used to enable custom authentication schemes e.g. test the value
187  * of a session cookie to determine if a user is logged on / authenticated
188  */
189  virtual bool execHttpHeaderValidation(String headerName, String headerValue) {
190  if(_httpHeaderValidationFunc) {
191  //return the value of the custom http header validation function
192  return _httpHeaderValidationFunc(headerName, headerValue);
193  }
194  //no custom http header validation so just assume all is good
195  return true;
196  }
